  def self.entry_to_path(entry)
    path = entry.to_s
    path = path.gsub('::', '/')
    path = path.gsub('#' , '/i-')
    path = path.gsub('.' , '/c-')
    path = OpEsc.escape(path)
    path = path + '.html'
  end

  def self.path_to_entry(path)
    entry = path.to_s
    entry = entry.chomp('.html')
    entry = OpEsc.unescape(entry)
    entry = entry.gsub('/c-' , '.')
    entry = entry.gsub('/i-' , '#')
    entry = entry.gsub('/'   , '::')
  end
